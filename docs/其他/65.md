        for skill_instance in &mut self.instances.skills {

迁移注册表的动态扩展
pub struct MigrationRegistry {
    migrations: Vec<(
        SemVer, 
        SemVer, 
        Box<dyn Fn(SaveData) -> Result<SaveData, MigrationError>>,
    )>,
}

impl MigrationRegistry {
    pub fn register<F>(&mut self, from: SemVer, to: SemVer, f: F)
    where
        F: Fn(SaveData) -> Result<SaveData, MigrationError> + 'static,
    {
        self.migrations.push((from, to, Box::new(f)));
        self.migrations.sort_by(|a, b| a.0.cmp(&b.0)); // 按版本排序
    }
    
    pub fn migrate(&self, data: SaveData, target: SemVer) -> Result<SaveData, MigrationError> {
        let mut current = data;
        while current.version < target {
            // 查找下一个迁移步骤
            let next_migration = self.migrations.iter()
                .find(|(from, _, _)| *from == current.version)
                .ok_or(MigrationError::UnsupportedVersion(current.version))?;
            
            current = (next_migration.2)(current)?;
        }
        Ok(current)
    }
}

这种设计支持动态注册迁移函数，便于MOD或插件扩展存档格式。

    // 属性范围验证
    for unit in &data.instances.units {
        if unit.attributes.current_hp < 0 {
            diagnostics.add_measurement("invalid_hp_count", 1.0);
            return Err(SaveValidationError::InvalidHp {
                unit_id: unit.id,
                hp: unit.attributes.current_hp,
            });
        }
    }
    
    // 性能监控
    let duration = start.elapsed().as_millis() as f64;
    diagnostics.add_measurement("save_validation_time", duration);
    
    Ok(())
}
